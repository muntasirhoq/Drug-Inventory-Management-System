CREATE TABLE COMPANY (

CompanyName VARCHAR2(20) NOT NULL,

CompanyID VARCHAR2(6),

Address VARCHAR2(20) NOT NULL,

Country VARCHAR2(15) NOT NULL,

Email VARCHAR2(30) NOT NULL,

Due NUMBER(6,0) NOT NULL,

PhoneNo VARCHAR2(11) NOT NULL,

PRIMARY KEY (CompanyID)


);


CREATE TABLE WORKER (

FirstName VARCHAR2(20) NOT NULL,

LastName VARCHAR2(20) NOT NULL,

WorkerID VARCHAR2(6) NOT NULL,

HireDate DATE NOT NULL,

Salary NUMBER(6,0) NOT NULL,

Address VARCHAR2(20) NOT NULL,

PhoneNo VARCHAR2(11) NOT NULL,

PRIMARY KEY (WorkerID) 

);

CREATE TABLE MEDICINE (

MedicineID VARCHAR2(6) NOT NULL,

MedicineName VARCHAR2(20) NOT NULL,

GenericName VARCHAR2(20) NOT NULL,

CompanyID VARCHAR2(6) NOT NULL,

RetailedPrice NUMBER(6) NOT NULL,

Status VARCHAR2(15) NOT NULL,

PRIMARY KEY (MedicineId),
FOREIGN KEY (CompanyID)
REFERENCES COMPANY(CompanyID)

);

CREATE TABLE SHIPMENTQUANTITY (

OrderNo VARCHAR2(6) NOT NULL,

MedicineID VARCHAR2(6) NOT NULL,

BatchNo VARCHAR2(6) UNIQUE,

Quantity NUMBER(6,0) NOT NULL,

PurchasedPrice NUMBER(6,0) NOT NULL,

EXPIRYDATE VARCHAR2(12) NOT NULL,

FOREIGN KEY (MedicineID)
REFERENCES MEDICINE (MedicineID)

);


CREATE TABLE SHIPMENT (
OrderNo VARCHAR2(6) NOT NULL,

CompanyID VARCHAR2(6) NOT NULL,

ShipDate VARCHAR2(12) NOT NULL,

PaidAmount NUMBER(6,0) NOT NULL,

WorkerId VARCHAR2(6) NOT NULL,

PRIMARY KEY (OrderNo),
FOREIGN KEY (WorkerID)
REFERENCES WORKER (WorkerID),
FOREIGN KEY (CompanyID)
REFERENCES COMPANY (CompanyID)
);


CREATE TABLE INVENTORY (

BatchNo VARCHAR2(6) NOT NULL,

MedicineID VARCHAR2(6) NOT NULL,

Quantity NUMBER(6,0) NOT NULL,

ExpiryDate VARCHAR2(12),

Primary Key (BatchNo)

);

CREATE TABLE CLIENT (

FirstName VARCHAR2(20) NOT NULL,

LastName VARCHAR2(20) NOT NULL,

ClientID VARCHAR2(6),

Due VARCHAR2 (7) NOT NULL,

Address VARCHAR2(20) NOT NULL,

PhoneNo VARCHAR2(11) NOT NULL,

PRIMARY KEY (ClientID) 

);

CREATE TABLE OLDCLIENT (

FirstName VARCHAR2(20) NOT NULL,

LastName VARCHAR2(20) NOT NULL,

ClientID VARCHAR2(6),

Due VARCHAR2 (7) NOT NULL,

Address VARCHAR2(20) NOT NULL,

PhoneNo VARCHAR2(11) NOT NULL,

PRIMARY KEY (ClientID) 

);

CREATE TABLE PURCHASEQUANTITY (

MedicineID VARCHAR2(6) NOT NULL,

BatchNo VARCHAR2(6) NOT NULL,

InvoiceNo VARCHAR2(6) NOT NULL,

Quantity NUMBER(6,0) NOT NULL,

FOREIGN KEY (BatchNo)
REFERENCES INVENTORY(BatchNo),
FOREIGN KEY (MedicineID)
REFERENCES MEDICINE(MedicineID)

);

CREATE TABLE PURCHASE (

InvoiceNo VARCHAR2(6) NOT NULL,

PurchasedDate DATE NOT NULL,

ClientId VARCHAR2(6) NOT NULL,

PaidAmount NUMBER(7,0) NOT NULL,

Discount NUMBER(4,0) NOT NULL,

WorkerID VARCHAR2(6) NOT NULL,

PRIMARY KEY (InvoiceNo),
FOREIGN KEY (WorkerId)
REFERENCES WORKER (WorkerID),
FOREIGN KEY(ClientID)
REFERENCES CLIENT(ClientID)

);
